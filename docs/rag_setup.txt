# ================================
# ARCHIVO: setup_milvus.sh
# ================================
# Script para configurar Milvus usando standalone_embed.sh

#!/bin/bash

echo "🚀 Configurando Milvus con standalone_embed.sh..."

# Descargar el script si no existe
if [ ! -f "standalone_embed.sh" ]; then
    echo "📥 Descargando standalone_embed.sh..."
    curl -sfL https://raw.githubusercontent.com/milvus-io/milvus/master/scripts/standalone_embed.sh -o standalone_embed.sh
    chmod +x standalone_embed.sh
fi

# Verificar que el script se descargó correctamente
if [ ! -f "standalone_embed.sh" ]; then
    echo "❌ Error: No se pudo descargar standalone_embed.sh"
    exit 1
fi

# Iniciar Milvus
echo "🔧 Iniciando Milvus..."
./standalone_embed.sh start

# Verificar que Milvus esté corriendo
echo "⏳ Esperando a que Milvus esté listo..."
sleep 10

# Test de conexión
echo "🔍 Verificando conexión a Milvus..."
if curl -s http://localhost:19530/health > /dev/null 2>&1; then
    echo "✅ Milvus está ejecutándose correctamente en puerto 19530"
else
    echo "⚠️ Milvus puede estar iniciando aún. Verifica con: curl http://localhost:19530/health"
fi

echo "✨ Configuración de Milvus completada!"
